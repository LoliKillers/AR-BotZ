```sh
$ pkg update -y
$ pkg upgrade -y
$ pkg install bash -y
$ pkg install git -y
$ pkg install nodejs -y
$ git clone https://github.com/LoliKillers/AR-BotZ
$ cd AR-BotZ
$ npm install
$ npm start/node index.js
```
